\question
Parametrization of linear paths with given points
\begin{parts}
    \part $0$ to $2+3i$
    \part $1+i$ to $2-i$
    \part $-1-i$ to $2+3i$
    \part $2$ to $2+2i$
\end{parts}   

\question
Parametrization of circular paths
\begin{parts}
    \part Circle centered at $0$ from $1$ to $i$
    \part Circle centered at $1+i$ with radius $2$. 
    \part Circle centered at origin of radius $3$.
\end{parts}

\question Find the parametrization of the curve $y=x^2$ from $0$ to $1$.

\question 
Find $\displaystyle \int_\gamma  (x^2-iy^2)dz $ along
\begin{parts}
    \part The parabola $y=2x^2$ from $(1,1)$ to $(2,8)$.
    \part The straight lines from $(1,1)$ to $(1,8)$ and $(1,1)$ to $(2,8)$
    \part The straight line from $(1,1)$ to $(2,8)$
\end{parts}
\question
Find $\displaystyle \int_\gamma \mid z \mid^2 dz$ along the square with vertices at $(0,0), (1,0), (1,1),(0,1)$.

\end{questions}
